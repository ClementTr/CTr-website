• Introduction

We have a .csv file 'data_processed.csv' containing the emails subjects of a company. Data are already cleaned.
1. The goal of this notebook will be to detect natural topics in those emails.
2. Then we will use this naturals topics to predict topic of each email.
3. Finally we want to draw links between employees according to a specific topic in order to know informal networks of a company.

from sklearn.metrics import silhouette_score

# It's a very long computation so we registered our results in a txt file
def get_silhouette(max_cluster, vectorized_param, min_cluster = 0):
    s = []
    if max_cluster <= min_cluster:
        return None
    for n_clusters in range(min_cluster ,max_cluster):
        kmeans = KMeans(n_clusters=n_clusters)
        kmeans.fit(vectorized_param)
        labels = kmeans.labels_
        s.append(silhouette_score(vectorized_param, labels, metric='euclidean'))
    return s

def plot_cluster_viz(num_clusters_viz, xs, ys, clusters_viz, num_title_words, max_colors=5):
    if num_clusters_viz > max_colors:
        print("Please add cluster_colors parameters")
        return 0
    else:
        cluster_viz_colors = {}
        cluster_viz_names = {}
        for i in range(num_clusters_viz):
            cluster_viz_colors[i] = cluster_colors[i]
            cluster_viz_names[i] = clusters_viz_title[i][:num_title_words]

        df = pd.DataFrame(dict(x=xs, y=ys, label=clusters_viz))
        groups = df.groupby('label')
        fig, ax = plt.subplots(figsize=(17, 9))
        for name, group in groups:
            ax.plot(group.x, group.y, marker='o', linestyle='', ms=12, label=cluster_viz_names[name],
                    color=cluster_colors[name], mec='none')
            ax.set_aspect('auto')
            ax.tick_params(axis= 'x', which='both', bottom='off', top='off', labelbottom='off')
            ax.tick_params(axis= 'y', which='both', left='off', top='off', labelleft='off')
        ax.legend(numpoints=1)
        return plt

from collections import Counter

def get_tag_used(column, n, s_left, s_right):
    subjects_list = []
    subjects = column.tolist()
    for s in subjects:
        s = s[s.find(s_left) + 1:s.find(s_right)]
        if s != "":
            subjects_list.append(s)
    key_words = Counter(subjects_list)
    return key_words.most_common(n)

def detect_tag(text, tag_list):
    for word in text.split():
        if word in tag_list:
            return 1
    return 0


def get_tag(df, all_tag):
    for k in all_tag.keys():
        column_dict = "Tag_" + str(k)
        if df[column_dict] == 1:
            return list(all_tag.keys()).index(k)
    return -1


def get_double_tag(df, col_topic_name, topic_tag):
    x = df[col_topic_name]
    topic_column = []
    for t in topic_tag:
        topic_column.append(str("Tag_" + t))
    for i in range(len(topic_tag)-1):
        if df[str(topic_column[i])] + df[str(topic_column[i+1])] == 2:
            return len(topic_tag) + i
    if df[str(topic_column[len(topic_tag)-1])] + df[str(topic_column[0])] == 2:
        return len(topic_tag)*2 - 1
    return x


def get_probs_tag(df_target, text_column, dic_all_tag):
    for k in dic_all_tag.keys():
        column_tag = "Tag_" + str(k)
        df_target[column_tag] = df_target[text_column].apply(detect_tag, args=(dic_all_tag[k],))
    df_target['Tag_Topic'] = df_target.apply(get_tag, args=(dic_all_tag,), axis=1)
    df_target['Tag_Topic'] = df_target.apply(get_double_tag, args=('Tag_Topic', list(dic_all_tag.keys())), axis=1)
    return df_target

import unicodedata

def clean_spec_char(text):
    text_clean = re.sub("-", " ", text)
    text_clean = text_clean.replace("?", " question ")
    text_clean = re.sub(r'[\w\.-]+@[\w\.-]+', ' ', text_clean)
    text_clean = re.sub("[^\w\d'\s]+", " ", text_clean)
    text_clean = re.sub(" CA ", " conseiladministration ", text_clean)
    text_clean = text_clean.lower()
    text_clean = re.sub(r"[a-z][']", " ", text_clean)
    text_clean = re.sub("'", " ", text_clean)
    text_clean = re.sub("ça", "cela", text_clean)
    text_clean = (unicodedata.normalize('NFKD', text_clean).encode('ascii', 'ignore')).decode("utf-8")
    # clean Version_1.1
    text_clean = re.sub("bis", " ", text_clean)
    text_clean = re.sub("devis", "expertise", text_clean)
    text_clean = re.sub("au revoir", "bye ", text_clean)
    # clean Version 2.0
    text_clean = re.sub(" o3 ", " oneonone ", text_clean)
    text_clean = re.sub("\d+", " ", text_clean)
    text_clean = re.sub(" pm ", "", text_clean)
    text_clean = re.sub(" am ", "", text_clean)
    return text_clean


def clean_from_spec_list(text, words_list):
    text_cleaned = ""
    for word in text.split():
        if word not in words_list:
            text_cleaned =  text_cleaned + " " + word
    return text_cleaned

        stopwords = f.read()
    stopwords_tokens = stopwords.split(separator)
    return stopwords_tokens

def clean_text(text, stopwords, hapax):
    text_cleaned = clean_spec_char(text)
    text_cleaned = clean_from_spec_list(text_cleaned, stopwords)
    text_cleaned = clean_from_spec_list(text_cleaned, hapax)
    text_cleaned = clean_plural(text_cleaned)
    return text_cleaned


def clean_plural(text):
    text_plural_cleaned = ""
    for word in text.split():
        word = word.rstrip('s')
        word = word.rstrip('x')
        text_plural_cleaned = text_plural_cleaned + " " + word
    return text_plural_cleaned

def get_interlocutors(x, y):
    z = [x, y]
    z.sort()
    z = z[0] + "|" + z[1]

get_my_network(df_target, sender_column, addressee_column, graph_name, generate_png=False, generate_gexf=False, top_n=None)
    'get_my_network' function from pyple package take as parameters a dataframe with 2 columns representing an exchange like sender-addressee.
    The second parameter will be the name of the first column, the third parameter will be the name of the second column.
    Using the fourth parameter you will give a name to your network as a file and as a plot graphic too.
    You can choose with the 5th and the 6th parameters to save the result as .png ou .gexf.
    If you don't specified these arguments, they won't be generated by default.
    The last parameter is optional and allow us to choose the number of the best interactions you want to work with.
    Without the 'top_n' argument, the function will work with the all interactions of your dataframe.
    Once this function will be executed, an array will be return with:
    - index 0: your dataframe concatenated
    - index 1: a python plot of your network
    - index 2: the final Graph value
