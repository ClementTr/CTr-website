In order to avoid this problem we will clean our 'The Hunger Games' text.

import re

def clean_text(x):

  # To convert the text in lowercase
  x = x.lower()
            
  # replace apostrophes
  x = x.replace("'", " ")
    
  # replace word where len <= 3
  x = re.sub('\\b\\w{1,3}\\s', '', x)
    
  # remove numbers/Digits
  x = re.sub(r'\d+', '', x)

  # remove white spaces
  x = " ".join(x.split())
  
  return(x)
  
len_text_before = len(text)
text = clean_text(text)
print("Size text before:", len_text_before)
print("Size text now:", len(text))
